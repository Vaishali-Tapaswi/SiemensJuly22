        package main

        import (
                "fmt"
                "strconv"
        )
        func add(x, y int) int{
                return x+y
        }

        func divide(x, y string) (int,error){
                no1, _ := strconv.Atoi(x)
                no2, _ := strconv.Atoi(y)
                if no2 == 0 {
                        return 0, fmt.Errorf("Divide Error because of zero second argument")
                }
                return no1/no2, nil
        }
